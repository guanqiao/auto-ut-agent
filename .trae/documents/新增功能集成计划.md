### Phase 2: Capability 重试集成 (优先级: 高)

#### T2.1 Capability 基类添加重试支持

**集成方案：**

```python
# 在 pyutagent/agent/capabilities/base.py 中添加
class Capability(ABC):
    def __init__(self):
        self._state: CapabilityState = CapabilityState.UNINITIALIZED
        self._retry_config: Optional[RetryConfig] = None
        self._retry_stats: Dict[str, RetryStats] = {}
    
    def get_retry_config(self) -> RetryConfig:
        """Get retry configuration for this capability."""
        if self._retry_config is None:
            self._retry_config = self._create_default_retry_config()
        return self._retry_config
    
    def _create_default_retry_config(self) -> RetryConfig:
        """Create default retry config. Override for customization."""
        return RetryConfig(max_attempts=3)
    
    def record_retry(self, operation: str, success: bool, attempts: int, 
                  error: Optional[Exception] = None):
        """Record retry attempt for statistics."""
        if operation not in self._retry_stats:
            self._retry_stats[operation] = RetryStats()
        stats = self._retry_stats[operation]
        stats.total_attempts += attempts
        if success:
            stats.success_count += 1
        else:
            stats.failure_count += 1
            if error:
                stats.errors.append(error)
    
    def get_retry_stats(self) -> Dict[str, Any]:
        """Get retry statistics for this capability."""
        return {
            name: self.name,
            stats: {k: v for k, v in self._retry_stats.items()},
            health: self.check_health(),
        }
    
    def check_health(self) -> Dict[str, Any]:
        """Check capability health based on retry stats."""
        stats = self._retry_stats
        if not stats:
            return {"healthy": True, "reason": "no_data"}
        
        total = sum(s.total_attempts for s in stats.values())
        failures = sum(s.failure_count for s in stats.values())
        
        if total == 0:
            return {"healthy": True}
        
        failure_rate = failures / total
        if failure_rate > 0.8:
            return {"healthy": False, "recommendation": "disable"}
        elif failure_rate > 0.5:
            return {"healthy": True, "recommendation": "monitor"}
        
        return {"healthy": True}
```

### Phase 3: StepExecutor 集成 (优先级: 高)

#### T3.1 StepExecutor 使用 Capability 重试配置

**集成方案：**

```python
# 在 pyutagent/agent/components/execution_steps.py 中

class StepExecutor:
    def __init__(self, agent_core, components, retry_config=None):
        # 从 Capability 获取重试配置（如果可用）
        self._capability_registry = components.get("capability_registry")
        if self._capability_registry:
            # 使用 Capability 系统的配置
            self.retry_config = self._get_capability_retry_config()
        else:
            # 使用传统配置
            self.retry_config = retry_config or DEFAULT_RETRY_CONFIG
    
    def _get_capability_retry_config(self) -> RetryConfig:
        """Get retry config from capability system."""
        # 合并所有 Capability 的重试配置
        configs = []
        for cap in self._capability_registry.get_all_ready():
            cap_config = cap.get_retry_config()
            if cap_config:
                configs.append(cap_config)
        
        if not configs:
            return DEFAULT_RETRY_CONFIG
        
        # 使用最严格的配置
        return RetryConfig(
            max_attempts=min(c.max_attempts for c in configs),
            backoff_base=max(c.backoff_base for c in configs),
            backoff_max=max(c.backoff_max for c in configs),
        )
```

### Phase 4: CapabilityRegistry 增强 (优先级: 中)

#### T4.1 自动健康检查

**集成方案：**

```python
# 在 pyutagent/agent/capabilities/registry.py 中添加
class CapabilityRegistry:
    def check_all_health(self) -> Dict[str, Dict[str, Any]]:
        """Check health of all capabilities."""
        return {
            name: cap.check_health()
            for name, cap in self._capabilities.items()
        }
    
    def auto_disable_unhealthy(self) -> List[str]:
        """Auto-disable unhealthy capabilities."""
        disabled = []
        for name, cap in self._capabilities.items():
            health = cap.check_health()
            if not health.get("healthy") and health.get("recommendation") == "disable":
                cap.disable()
                disabled.append(name)
                logger.warning(f"Auto-disabled unhealthy capability: {name}")
        return disabled
    
    def get_retry_stats(self) -> Dict[str, Any]:
        """Get retry statistics from all capabilities."""
        return {
            name: cap.get_retry_stats()
            for name, cap in self._capabilities.items()
            if cap.is_ready
        }
```

from pyutagent.core.metrics import get_metrics

class CapabilityRegistry:
    def __init__(self, container=None):
        self._metrics = get_metrics()
    
    def load_all(self, config=None) -> Dict[str, bool]:
        results = super().load_all(config)
        
        # 记录指标
        success_count = sum(1 for v in results.values() if v)
        self._metrics.record("capability_load", {
            "total": len(results),
            "success": success_count,
            "failure": len(results) - success_count,
        })
        
        return results
```

# 在 pyutagent/agent/enhanced_agent.py 中添加
class EnhancedAgent(ReActAgent):
    def get_capability_retry_stats(self) -> Dict[str, Any]:
        """Get retry statistics from all capabilities."""
        if not self.capability_registry:
            return {}
        return self.capability_registry.get_retry_stats()
    
    def get_enhanced_stats(self) -> Dict[str, Any]:
        stats = super().get_enhanced_stats()
        
        # 添加 Capability 重试统计
        if self.capability_registry:
            stats["capability_retry_stats"] = self.get_capability_retry_stats()
            stats["capability_health"] = self.capability_registry.check_all_health()
        
        return stats
```
