## 🚀 集成指南

### 1. 在 ReActAgent 中集成

```python
from pyutagent.agent.intelligence_enhancer import IntelligenceEnhancer
from pyutagent.llm.intelligence_cot import IntelligenceEnhancedCoT

class ReActAgent:
    def __init__(self, ...):
        # 添加智能化组件
        self.intelligence_enhancer = IntelligenceEnhancer()
        self.intelligence_cot = IntelligenceEnhancedCoT()
    
    def generate_tests(self, file_path: str, java_class: Any):
        # 1. 智能分析
        intelligence_result = self.intelligence_enhancer.analyze_target_code(
            file_path, java_class
        )
        
        # 2. 生成增强提示词
        base_prompt = self.prompt_builder.build_prompt(java_class)
        enhanced_prompt = self.intelligence_enhancer.generate_enhanced_prompt(
            base_prompt, intelligence_result
        )
        
        # 3. 调用 LLM
        test_code = self.llm_client.generate(enhanced_prompt)
        
        return test_code
    
    def fix_errors(self, error_output: str, test_code: str, source_code: str):
        # 1. 根因分析
        analysis = self.root_cause_analyzer.analyze_test_failures(
            error_output, test_code, source_code
        )
        
        # 2. 生成增强修复提示
        fix_prompt = self.intelligence_cot.generate_enhanced_fix_prompt(
            error_output, test_code, source_code, "testMethod"
        )
        
        # 3. 调用 LLM 修复
        fixed_code = self.llm_client.generate(fix_prompt)
        
        return fixed_code
```

## 💡 使用示例

### 示例 1: 代码语义分析

```python
from pyutagent.core.semantic_analyzer import SemanticAnalyzer

# 创建分析器
analyzer = SemanticAnalyzer()

# 解析 Java 文件
java_class = java_parser.parse_file("UserService.java")

# 执行分析
result = analyzer.analyze_file("UserService.java", java_class)

# 获取测试场景
scenarios = analyzer.get_test_scenarios()
for scenario in scenarios:
    print(f"场景：{scenario.description}")
    print(f"类型：{scenario.test_type}")
    print(f"优先级：{scenario.priority}")

# 获取边界条件
boundaries = result.get("boundary_conditions", [])
for boundary in boundaries:
    print(f"参数：{boundary['parameter']}")
    print(f"类型：{boundary['type']}")
    print(f"测试值：{boundary['test_value']}")
```

### 示例 2: 错误根因分析

```python
from pyutagent.core.root_cause_analyzer import RootCauseAnalyzer

# 创建分析器
analyzer = RootCauseAnalyzer()

# 分析编译错误
compiler_output = """
[ERROR] /path/to/File.java:10: error: ';' expected
[ERROR] /path/to/File.java:15: error: incompatible types
"""

analysis = analyzer.analyze_compilation_errors(compiler_output)

# 查看根因
for cause in analysis.root_causes:
    print(f"根因：{cause.description}")
    print(f"类别：{cause.category.name}")
    print(f"置信度：{cause.confidence:.0%}")

# 查看推荐修复
for fix in analysis.suggested_fixes:
    print(f"策略：{fix.description}")
    print(f"优先级：{fix.priority}")
    print(f"成功率：{fix.success_probability:.0%}")
```

- ✅ **失败后分析**: 在测试失败后执行根因分析
- ✅ **缓存结果**: 避免重复分析同一文件

### 2. 决策策略
```python
# 基于置信度做决策
if analysis.confidence_score > 0.7:
    # 高置信度 - 自动修复
    apply_fix(analysis.suggested_fixes[0])
elif analysis.confidence_score > 0.4:
    # 中等置信度 - 尝试修复
    retry_with_strategy(analysis.suggested_fixes[0])
else:
    # 低置信度 - 请求人工审查
    request_human_review(analysis)
```
